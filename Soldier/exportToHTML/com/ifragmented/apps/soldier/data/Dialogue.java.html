24       public Dialogue(String body, Answer[] answers, DialogueType dialogueType, IActor actor) { 
25           this.body = body; 
26           this.answers = answers; 
27           this.actor = actor; 
28           this.dialogueType = dialogueType;

33       public Dialogue(String body, Answer[] answers, DialogueType dialogueType, IActor actor, DifficultyLevel difficultyLevel) { 
34           this.body = body; 
35           this.answers = answers; 
36           this.actor = actor; 
37           this.dialogueType = dialogueType; 
38           this.isAnswered = false; 
39           this.difficultyLevel = difficultyLevel; 
40       } 
41    
42       public String getBody() { 
43           return body; 
44       } 
45    
46       public Answer[] getAnswers() { 
47           return answers.clone(); 
48       } 
49    
50       public void answer(String userAnswer){ 
51           for(Answer a : answers){ 
52               if(a.getAnswerText().equalsIgnoreCase(userAnswer)){ 
53                   chosenAnswer = a; 
54                   isAnswered = true; 
55                   break; 
56               } 
57           } 
58           //should throw error if no correct answer given to avoid program hanging forever 
59           if(chosenAnswer==null){ 
60               throw new RuntimeException("No matching answer has been given."); 
61           } 
62       }

66       }
