97       public Story getRandomStory(int MAX_ENCOUNTERS) { 
98           SQLiteDatabase db= this.getReadableDatabase(); 
99           Log.d("DBHelper","Getting random story"); 
100   
101          String randomQuery = "SELECT * FROM "+ StoryTableFeed.TABLE_NAME+" ORDER BY RANDOM() LIMIT 1;"; 
102   
103          String storyId; 
104          String storyTitle; 
105          String storyIntroId; 
106          String storyOutroL; 
107          String storyOutroW; 
108          String storyBkgrnd; 
109   
110          Cursor randomRowCursor = db.rawQuery(randomQuery,null); 
111          randomRowCursor.moveToFirst(); 
112          storyId = randomRowCursor.getString(1); 
113          storyTitle = randomRowCursor.getString(2); 
114          storyIntroId = randomRowCursor.getString(3); 
115          storyOutroL = randomRowCursor.getString(4); 
116          storyOutroW = randomRowCursor.getString(5); 
117          storyBkgrnd = randomRowCursor.getString(6); 
118   
119          Log.d("DBHelper","Random story lotted, info: " + storyId +" " + storyTitle + " intro:" 
120                  + storyIntroId + " outroL" + storyOutroL + " outroW:" + storyOutroW 
121                  + " and background image: " + storyBkgrnd); 
122   
123          randomRowCursor.close(); 
124   
125          Dialogue intro = fetchDialogueFromDB(storyIntroId); 
126          Dialogue outroL = fetchDialogueFromDB(storyOutroL); 
127          Dialogue outroW = fetchDialogueFromDB(storyOutroW); 
128          ArrayList<Dialogue>storyDialogues= getDefaultDialoguesForStory(storyId,MAX_ENCOUNTERS); 
129   
130          Story story = new Story(storyTitle,intro,outroL,outroW,storyDialogues,storyBkgrnd); 
131          Log.d("DBHelper","\nDone creating random story for game\n"); 
132          db.close(); 
133          return story; 
134      }

190      private Dialogue fetchDialogueFromDB(String dialogueId){ 
191          if(dialogueId==null){return null;} 
192          Dialogue dialogue = null; 
193          SQLiteDatabase db= this.getReadableDatabase(); 
194          String randomQuery = "SELECT * FROM "+ DialogueTableFeed.TABLE_NAME+" WHERE "+DialogueTableFeed.COLUMN_NAME_ID + " = '" + dialogueId+"' ;"; 
195          Log.d("DBHELPER","Searching for dialogue with id: " + dialogueId + " and by sql: \n" + randomQuery); 
196          Cursor dialogueCursor = db.rawQuery(randomQuery,null); 
197          if(dialogueCursor==null || !dialogueCursor.moveToNext()){return null;} 
198          String actor = dialogueCursor.getString(1); 
199          String body = dialogueCursor.getString(2); 
200          String id = dialogueCursor.getString(3); 
201          String type = dialogueCursor.getString(4); 
202          String diffLvl = dialogueCursor.getString(5); 
203          String parentStory = dialogueCursor.getString(6); 
204          int triggerValue = dialogueCursor.getInt(7); 
205          Log.d("DBHELPER","Fetched dialogue with data:" +actor+" " + body+" " + id+" " + type+" "+ "and parent story:" + parentStory); 
206          dialogueCursor.close(); 
207          Answer [] answers = fetchAnswersFromDB(id); 
208          Actor dialActor = fetchActorFromDB(actor); 
209          DialogueType dt = DialogueType.valueOf(type); 
210          DifficultyLevel df = DifficultyLevel.getDifficulty(Integer.parseInt(diffLvl)); 
211          dialogue = new Dialogue(body,answers,dt,dialActor,df); 
212          dialogue.setTrigger(triggerValue == 1); 
213          dialogueCursor.close(); 
214          db.close(); 
215          return dialogue; 
216      }

218      private Answer[] fetchAnswersFromDB(String id) { 
219          if(id==null){return null;} 
220          Answer[]answers=null; 
221          SQLiteDatabase db= this.getReadableDatabase(); 
222          String randomQuery = "SELECT * FROM "+ AnswerTableFeed.TABLE_NAME+" WHERE "+AnswerTableFeed.COLUMN_PARENT_DIALOGUE + " = '" + id+"' ;"; 
223          Log.d("DBHELPER","Searching for answers for dialogue with id: " + id + " and by sql: \n" + randomQuery); 
224          Cursor dialogueCursor = db.rawQuery(randomQuery,null); 
225          if(dialogueCursor==null || dialogueCursor.getCount()<1){ 
226              Log.d("DBHelper","Error, row count 0 for " + id); 
227              return null; 
228          } 
229          ArrayList<Answer>answersList = new ArrayList<>(); 
230          while (dialogueCursor.moveToNext()){ 
231              String answerID = dialogueCursor.getString(1); 
232              String answeBody=dialogueCursor.getString(2); 
233              String outcome=dialogueCursor.getString(3); 
234              String followup=dialogueCursor.getString(4); 
235              String difficultyOffset =dialogueCursor.getString(5); 
236              String answerParent = dialogueCursor.getString(6); 
237              Dialogue followupD = null; 
238              if(followup!=null && !followup.isEmpty()){ 
239                  followupD = fetchDialogueFromDB(followup); 
240   
241              } 
242              Answer answer = new Answer(answerID,answeBody,Outcome.valueOf(outcome), DifficultyOffset.offsetForValue(difficultyOffset),followupD); 
243              answersList.add(answer); 
244          } 
245          answers= new Answer[answersList.size()]; 
246          for(int i=0;i<answers.length;i++){ 
247              answers[i]=answersList.get(i); 
248          } 
249          dialogueCursor.close(); 
250          db.close(); 
251          return answers; 
252      }

256          SQLiteDatabase db= this.getReadableDatabase(); 
257          String randomQuery = "SELECT * FROM "+ ActorTableFeed.TABLE_NAME+" WHERE "+ActorTableFeed.COLUMN_ACTOR_ID + " = '" + actor+"' ;";

263          } 
264          if (dialogueCursor.moveToNext()) {  //should be only 1 result 
265              String id =dialogueCursor.getString(1); 
266              String name = dialogueCursor.getString(2); 
267              String img = dialogueCursor.getString(3); 
268              a = new Actor(id,name,img);

270          }

282      public List<Challenge> getChallenges(GameSettings gameSettings){ 
283          SQLiteDatabase db= this.getReadableDatabase(); 
284          Log.d("DBHelper","Getting challenges."); 
285          String[]scoreProjection ={ChallengeTableFeed._ID, 
286                  ChallengeTableFeed.COLUMN_CH_BODY, 
287                  ChallengeTableFeed.COLUMN_CH_ID, 
288                  ChallengeTableFeed.COLUMN_CH_DIFFICULTY, 
289                  ChallengeTableFeed.COLUMN_CH_ALTERNATIVES, 
290                  ChallengeTableFeed.COLUMN_CH_CORRECT}; 
291   
292          Cursor cursor = db.query( 
293                  ChallengeTableFeed.TABLE_NAME,                // The table to query 
294                  scoreProjection,                                                // The columns to return 
295                  null,                                                           // The columns for the WHERE clause 
296                  null,                                                           // The values for the WHERE clause 
297                  null,                                                           // don't group the rows 
298                  null,                                                           // don't filter by row groups 
299                  null                                                            // The sort order 
300          ); 
301   
302          ArrayList<Challenge>challenges = new ArrayList<>(); 
303          while(cursor.moveToNext()){ 
304              //Log.d("PrintOfChallenge",cursor.get); 
305              String body = cursor.getString(1); 
306              String correct = cursor.getString(5); 
307              String alt = cursor.getString(4); 
308              String diff = cursor.getString(3); 
309              String id = cursor.getString(2); 
310              String [] answers= (String[]) divideChallengeAnswers(alt).toArray(); 
311              DifficultyLevel difficultyLevel = DifficultyLevel.valueOf(diff); 
312              Challenge challenge = new Challenge(body,answers,correct,difficultyLevel); 
313              challenge.setChallengePoints(new ChallengePoints(gameSettings,difficultyLevel)); 
314              challenges.add(challenge); 
315          } 
316          cursor.close(); 
317   
318          return challenges; 
319      }
